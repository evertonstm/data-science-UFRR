import pandas as pd
import matplotlib.pyplot as plt

# Calcular a taxa de mortalidade
base_combinada['Taxa de Mortalidade'] = (base_combinada['mortes'] / base_combinada['confirmadas']) * 100

# Encontrar o país com a maior e menor taxa de mortalidade
pais_maior_taxa = base_combinada.loc[base_combinada['Taxa de Mortalidade'].idxmax(), 'País']
maior_taxa = base_combinada['Taxa de Mortalidade'].max()

pais_menor_taxa = base_combinada.loc[base_combinada['Taxa de Mortalidade'].idxmin(), 'País']
menor_taxa = base_combinada['Taxa de Mortalidade'].min()

print(f"País com a maior taxa de mortalidade: {pais_maior_taxa} - Taxa: {maior_taxa:.2f}%")
print(f"País com a menor taxa de mortalidade: {pais_menor_taxa} - Taxa: {menor_taxa:.2f}%")

# Calcular a taxa de letalidade
base_combinada['Taxa de Letalidade'] = (base_combinada['mortes'] / base_combinada['confirmadas']) * 100

# Encontrar o país com a maior e menor taxa de letalidade
pais_maior_letalidade = base_combinada.loc[base_combinada['Taxa de Letalidade'].idxmax(), 'País']
maior_letalidade = base_combinada['Taxa de Letalidade'].max()

pais_menor_letalidade = base_combinada.loc[base_combinada['Taxa de Letalidade'].idxmin(), 'País']
menor_letalidade = base_combinada['Taxa de Letalidade'].min()

print(f"País com a maior taxa de letalidade: {pais_maior_letalidade} - Taxa: {maior_letalidade:.2f}%")
print(f"País com a menor taxa de letalidade: {pais_menor_letalidade} - Taxa: {menor_letalidade:.2f}%")

# Escolher os países para plotar os gráficos
paises_maior_mortalidade = [pais_maior_taxa, pais_menor_taxa]
paises_maior_letalidade = [pais_maior_letalidade, pais_menor_letalidade]

# Filtrar o dataframe para os países escolhidos
dados_mortalidade = base_combinada[base_combinada['País'].isin(paises_maior_mortalidade)]
dados_letalidade = base_combinada[base_combinada['País'].isin(paises_maior_letalidade)]

# Plotar os gráficos de linha
plt.figure(figsize=(10, 5))
for pais in paises_maior_mortalidade:
    plt.plot(dados_mortalidade.columns[1:-2], dados_mortalidade.loc[dados_mortalidade['País'] == pais].values[0][1:-2], label=pais)
plt.title('Taxa de Mortalidade ao Longo do Tempo')
plt.xlabel('Data')
plt.ylabel('Taxa de Mortalidade (%)')
plt.legend()
plt.xticks(rotation=45)
plt.grid(True)
plt.tight_layout()
plt.show()

plt.figure(figsize=(10, 5))
for pais in paises_maior_letalidade:
    plt.plot(dados_letalidade.columns[1:-2], dados_letalidade.loc[dados_letalidade['País'] == pais].values[0][1:-2], label=pais)
plt.title('Taxa de Letalidade ao Longo do Tempo')
plt.xlabel('Data')
plt.ylabel('Taxa de Letalidade (%)')
plt.legend()

plt.grid(True)
